--QUERY1

SELECT a.department_name, 
       b.first_name, 
       b.last_name 
FROM L_DEPARTMENTS a,
     L_EMPLOYEES b 
WHERE a.dept_code = b.dept_code;

--QUERY1B

SELECT a.department_name, 
       b.first_name, 
       b.last_name 
FROM L_DEPARTMENTS a 
     LEFT OUTER JOIN L_EMPLOYEES b 
        ON a.dept_code = b.dept_code;
